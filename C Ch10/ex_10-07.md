/* prog10_7, 傳遞指標到函數裡 */
#include <stdio.h>
#include <stdlib.h>
void address(int *);      /* 宣告指標變數型態(void->不傳回值) */
int main(void)
{
   int a=12; 			    /* 設定變數a的值為12 */
   int *ptr=&a; 			 /* 將指標ptr指向變數a */
   
   address(&a);			/* 將a的位址當參數傳給address()函數 */
   address(ptr); 		   /* 將ptr傳入address()函數中 */
   printf("最後a的值(value)為%d\n,a");
   
   system("pause");
   return 0;
}
void address(int *p1)
{
   printf("於位址%p內，儲存的變數內容為%d\n",p1,*p1);  /* 到p1的位址取值 */ 
   *p1=*p1+5;
}
