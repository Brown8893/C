### 一般函數在main函數之前的不須要宣告
```
#include <stdio.h>

/* writeString函數: 顯示整個程式的功能*/ 
void writeString(void)
{
   printf("歡迎使用C!\n"); 
   printf("本程式的功能是計算1加到100 \n"); 
   printf("本程式的功能是由超級賽雅人所設計 \n");
}

/* one2many函數: 顯示1加到100 */ 
void one2many()
{
    /* 變數宣告 */
    int i;
    int total = 0;
    /* 迴圈敘述 */
    for ( i = 1; i <= 100; i++ )
    {
        printf("數字: %d\n", i);
        total += i;
    }
    printf("從1到100的總和: %d\n", total);
}

/* 主程式 */ 
main()
{
    /* 函數的呼叫 */
    writeString();
    one2many();
    getchar();
}
```

### 一般函數在main函數之後的必須要先宣告
```
#include <stdio.h>

/* 函數的原型宣告 */
int n2N(int, int);

/* writeString函數: 顯示整個程式的功能*/ 
void writeString(void)
{
   printf("歡迎使用C!\n"); 
   printf("本程式的功能是計算1加到100 \n"); 
   printf("本程式的功能是由超級賽雅人所設計 \n");
}

/* 主程式 */ 
main()
{
/* 變數宣告 */
  int total, start, end;
  char c;

writeString();
  do{
    printf("請輸入範圍n_start n_final: ");
    scanf("%d%d", &start, &end);

    /* 函數的呼叫 */
    total = n2N(start, end);
  
  printf("從%d到%d的總和: %d\n", start, end, total);
    
/* 讀取剩下的字元 */
    while ( (c=getchar()) != '\n' );
    } while( start != -1 ); 
}
/* 函數: 計算start到end的數字和 */ 
int n2N(int start, int end)
{
    /* 變數宣告 */
    int i;
    int total = 0;
    /* 迴圈敘述 */
    for ( i = start; i <= end; i++ )
        total += i;
    return total;
}
```

```
double power_for(double base, int n) 
{
   int i;
   double pow=1.0;
   for(i=1;i<=n;i++)		
      pow=pow*base;
   return pow;
}
```
