## Ans_score[4]={99,11,23,44}
```
#include <stdio.h>
#include <stdlib.h>
int main(void)
{
   int i,score[4]; 	/* 宣告整數變數i與整數陣列score */
   
   score[0]=99;		/* 設定陣列的第一個元素為78 */
   score[1]=11; 	/* 設定陣列的第二個元素為55 */
   score[2]=23;   /* 設定陣列的第三個元素為92 */
   score[3]=44; 	/* 設定陣列的最後一個元素為80 */
   
   for(i=0;i<=3;i++)
      printf("score[%d]=%d\n",i,score[i]);  /* 印出陣列的內容 */
   
   system("pause");
   return 0;
}
```
